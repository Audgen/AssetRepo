pipeline {
    agent any
    stages {
    	stage('Initialize'){
            steps {
                sh 'chmod -R 777 ${WORKSPACE}'
            }
        }
	stage('Create base docker file'){
            steps {
                sh "$sudo {MSR_TARGET_DOCKER}/is_container.sh createDockerfile"
            }
        }
	stage('Build base MSR image'){
            steps {
                sh "$sudo {MSR_TARGET_DOCKER}/is_container.sh build -Dimage.name=is:msr"
            }
        }
     }
    }
