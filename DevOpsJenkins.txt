pipeline {
    agent any 
    stages {
        stage('Initialize'){
            steps {
                sh 'chmod -R 777 ${WORKSPACE}'
            }
        }
	stage('Build'){
            steps {
		sh "/opt/softwareag/common/AssetBuildEnvironment/bin/build.sh"
            }
        }
	stage('Cleanup'){
            steps {
                sh 'chmod -R 777 /opt/CI/builds/'
            }
        }
        stage('CreateProject') {
            steps {
	    	sh "/opt/softwareag103/IntegrationServer/instances/default/packages/WmDeployer/bin/projectautomatorUnix.sh ${WORKSPACE}/ProjectAutomator_Full.xml"
            }
        }
	stage('Deploy') {
            steps {
		sh "/opt/softwareag103/IntegrationServer/instances/default/packages/WmDeployer/bin/Deployer.sh --deploy -dc SDIS_DEPLOYMENTCANDIDATE -project SDIS_PROJECT -host vmdexcel.eur.ad.sag -port 5555 -user Administrator -pwd manage"
            }
        }
 	stage('Test') {
		steps {
		    sh "Scripts/Test.sh"
			junit 'reports/*.xml'
            }
        }
    }
}
